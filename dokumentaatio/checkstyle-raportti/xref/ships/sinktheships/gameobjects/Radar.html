6   package ships.sinktheships.gameobjects;
7   
8   import java.util.Random;
9   
10  /**
11   * Luokan tehtävä on tarvittaessa näyttää pelaajan laivan perän sijainti
12   * koordinaatistossa. Pelaajan tulittaessa laivallaan tämä luokka arpoo
13   * näytetäänkä tulittaneen laivan sijainti. Tähän liittyy satunnaisuutta, jos
14   * tulittaa vain kerran, ei koskaan näytetä, jos kuudesti, näytetään aina ja
15   * muuten se on sattumasta kiinni.
16   *
17   * @author jambo
18   */
19  public class Radar {
20  
21      private Coordinate shipLocation;
22      private int timesShipHasShot = 0;
23      private Ship ship;
24      private boolean active;
25      private boolean isVisible = false;
26  
27      /**
28       * Tutkalle annetaan laiva. Tutka aktivoituu ja tallettaa laivan ja sen
29       * perän sijainnin.
30       *
31       * @param ship annettu laiva
32       */
33      public Radar(Ship ship) {
34          this.active = true;
35          this.ship = ship;
36          this.shipLocation = this.ship.getSternCoordinate();
37      }
38  
39      /**
40       * Vaihtoehtoinen konstruktori, jota kutsutaan ennen kuin pelaajilla on
41       * vielä laivoja. Tarkoituksena vähentää if-operaatioita muista luokista.
42       */
43      public Radar() {
44          this.active = false;
45      }
46  
47      /**
48       * Palauttaa tutkalle tallennetun jonkin laivan perän sijainnin.
49       *
50       * @return koordinaatti, laivan perän sijainti
51       */
52      public Coordinate getCoordinate() {
53          return this.shipLocation;
54      }
55  
56      /**
57       * Käytetään kun laivalla on tulitettu, lisää todennäköisyyttä, että tutkaan
58       * talletettu laiva tulee nähdyksi.
59       */
60      public void shipHasShot() {
61          this.timesShipHasShot += 1;
62          updateVisible();
63      }
64  
65      private void updateVisible() {
66          if (!isVisible) {
67              int randomInt = getRandomNumberInRange(this.timesShipHasShot - 1, 5);
68              long randomLong = randomInt / 5; //löytyy väliltä 0-1
69  
70              System.out.println("updating visible");
71              isVisible = randomLong >= 0.5;
72          }
73      }
74  
75      /**
76       * Kertoo näkyykö laiva tutkassa.
77       *
78       * @return boolean, tosi jos laiva näkyy
79       */
80      public boolean thereIsRadarImage() {
81          return isVisible;
82      }
83  
84      private int getRandomNumberInRange(int min, int max) {
85          Random rand = new Random();
86          return rand.nextInt((max - min) + 1) + min;
87      }
88  
89      /**
90       * Kertoo onko tutka aktiivinen, eli onko siihen talletettu laiva.
91       *
92       * @return boolean, tosi kun tutkaan talletettu laiva
93       */
94      public boolean isActive() {
95          return active;
96      }
97  
98  }
